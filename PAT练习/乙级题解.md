```c++
void reverseNum(vector<int>& st,int s,int e)
{
    for(int i = 0;i < (e - s + 1)/2;++i)
    {
        int temp = st[i + s];
        st[i + s] = st[e - i];
        st[e - i] = temp;
    }
```

### **1015 德才论**

> 考点：排序
>
> 总结：注意分类，解题过程中可以将要点打出来

```c++
/*
L 最低分数线 H为优先录取J
(1 不低于H 的 才德全尽   2 才分不到H但得分到的  德胜才 
3 德才均低于H 但是德分 不低于才分  4 其他的后面       )

分数从高到低排序 总分相同 按照德分降序  否则 按照id升序
*/
#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;
struct node
{
    long long id;
    int de,cai;
    bool operator<(const node &e)const
    {
        if(de + cai  != e.de + e.cai)
        {
            return de + cai > e.de + e.cai;
        }
        else 
        {
            if(de != e.de)
            {
                return de > e.de;
            }
            else 
            {
                return id < e.id;
            }
        }
    }
};
int N,L,H;
vector<node> list[4];


int main()
{
    cin>>N>>L>>H;
    int ans = 0;
    for(int i = 0;i < N;++i)
    {
        long long a;
        int b,c;
        cin>>a>>b>>c;
        if(b >= L && c >= L)
        {
            ++ans;
            if(H <= b && H <= c)
            {
                list[0].push_back({a,b,c});
            }
            else if(H <= b && H > c)
            {
                list[1].push_back({a,b,c});
            }
            else if(H > b && H > c && b >= c)
            {
                list[2].push_back({a,b,c});
            }
            else 
            {
                list[3].push_back({a,b,c});
            }
        }
    }

    for(int i = 0;i < 4;i++)
    {
        sort(list[i].begin(),list[i].end());
    }

    cout<<ans<<endl;
    for(int i = 0;i < 4;i++)
    {
        for(auto &e : list[i])
```

### **1073 多选题常见计分法** 

> 考点：
>
> 总结：利用异或运算。注意求每道题每个选项的错误次数利用二维数组存储。`scanf`读取输入内容的格式。

```c++
#include<iostream>
#include<vector>
#include<cmath>

using namespace std;

int main()
{
    int n,m,optnum,truenum,temp,maxcnt = 0;
    //a b c d e 分别用二进制0001 0010 0100 1000 10000代替
    int hash[] = {1,2,4,8,16},opt[1010][110] = {0};
    char c;
    scanf("%d %d",&n,&m);
    vector<int> fullscore(m),trueopt(m);
    vector<vector<int>> cnt(m,vector<int>(5));

    for(int i = 0;i < m;i++)
    {
        scanf("%d %d %d",&fullscore[i],&optnum,&truenum);
      
        for(int j = 0;j < truenum;j++)
        {

            scanf(" %c",&c);
            trueopt[i] += hash[c - 'a'];
        }
    }

    for(int i = 0;i < n;i++)
    {
        double grade = 0;
        for(int j = 0;j < m;j++)
        {
            getchar();
            scanf("(%d",&temp);
            for(int k = 0;k < temp;k++)
            {
                if(k == temp - 1)
                    scanf(" %c)",&c);
                else 
                    scanf(" %c",&c);

                // scanf(" %c)",&c);
                opt[i][j] += hash[c - 'a'];
            }
    
            int el = opt[i][j] ^ trueopt[j];
       
            if(el)
            {
                //判断没有错误选项
                if((opt[i][j] | trueopt[j]) == trueopt[j]){
                   
                    grade += fullscore[j] * 1.0 / 2;
                }
                if(el)
                {//记录记录错误的题（这道题的各个选项错误的次数）
                    for(int k = 0;k < 5;k++)
                        if(el & hash[k]) cnt[j][k]++;
                }
            } else {
           
                grade += fullscore[j];
            }
        }
        printf("%.1f\n",grade);
    }
    for(int i = 0;i < m;i++)
        for(int j = 0;j < 5;j++)
            maxcnt = maxcnt > cnt[i][j] ? maxcnt : cnt[i][j];
    if(maxcnt == 0)
    {
        printf("Too simple\n");
    }
    else 
    {
        //指出哪道题的哪个选项错的人最多
        for(int i = 0;i < m;i++)
        {
            for(int j = 0;j < cnt[i].size();j++)
            {
                if(maxcnt == cnt[i][j])
                    printf("%d %d-%c\n",maxcnt,i + 1,'a' + j);
            }
        }
    }

    return 0;
}
```

```c++
	for(int k = 0;k < n;k++)
	{
		if(v[i] - dif * k < 0 || prime[v[i] - dif * k])
			return false;
	}
```
